# Define a cleaning function to handle all types of non-numeric characters
def clean_numeric_column(df, column_name):    
    # Convert data to strings
    series = df[column_name].astype(str)
    
    # Replace text numbers
    number_map = {'Two': '2', 'Twenty': '20', 'Hundred': '100'}
    for word, num in number_map.items():  
        series = series.str.replace(word, num)
    
    # Handle $ symbol, spaces, and NAN values
    series = series.str.replace('$', '').str.strip()
    series = series.replace(['nan', 'NAN', '*', '***', '****'], pd.NA)
    
    df[column_name] = pd.to_numeric(series, errors='coerce')
    return df

Step 1: Outlier Detection and Analysis¶

  1. Statistical Analysis

  2. IQR-based Outlier Detection

    • Q1 (25th percentile) and Q3 (75th percentile)
    • Interquartile Range (IQR = Q3 - Q1)
    • Outlier boundaries: Lower = Q1 - 1.5×IQR, Upper = Q3 + 1.5×IQR
    • Count and calculate percentage of outliers beyond these boundaries

  3. Visual Analysis

    • Generate box plots to visualize data distribution and outlier positions
    • Create scatter plots for correlated variables (AIR_TIME vs DISTANCE)

  4. Correlation Assessment

    • Calculate correlation coefficients between related variables
    • Validate logical relationships (e.g., longer distances should correlate with longer air times)

def check_outliers_comprehensive(df, columns):
    for col in columns:
        print(f"\n{'='*50}")
        print(f"Checking outliers for {col}")
        print(f"{'='*50}")
        
        # 1. Basic statistics
        print("\n1. Basic Statistics:")
        stats = df[col].describe()
        print(stats)
        
        # 2. Box plot analysis
        Q1 = stats['25%']
        Q3 = stats['75%']
        IQR = Q3 - Q1
        lower_bound = Q1 - 1.5 * IQR
        upper_bound = Q3 + 1.5 * IQR
        
        outliers = df[(df[col] < lower_bound) | (df[col] > upper_bound)][col]
        print(f"\n2. Box Plot Rule Outliers:")
        print(f"Lower bound: {lower_bound:.2f}")
        print(f"Upper bound: {upper_bound:.2f}")
        print(f"Number of outliers: {len(outliers)}")
        print(f"Outlier percentage: {(len(outliers)/len(df)*100):.2f}%")
        
        plt.figure(figsize=(10, 4))
        sns.boxplot(x=df[col])
        plt.title(f'Box Plot for {col}')
        plt.show()
        
        # 3. Correlation check (for specific columns)
        if col in ['AIR_TIME', 'DISTANCE'] and 'AIR_TIME' in df.columns and 'DISTANCE' in df.columns:
            correlation = df['AIR_TIME'].corr(df['DISTANCE'])
            print(f"\n3. Correlation between AIR_TIME and DISTANCE: {correlation:.2f}")
            
            plt.figure(figsize=(10, 6))
            plt.scatter(df['DISTANCE'], df['AIR_TIME'], alpha=0.1)
            plt.xlabel('DISTANCE')
            plt.ylabel('AIR_TIME')
            plt.title('AIR_TIME vs DISTANCE')
            plt.show()

Step 2: Outlier Handling¶

Outlier Removal Summary:

  • Air_Time

Negative values: physically impossible flight times

600 minutes: unrealistic for US domestic flights (max should be ~8-9 hours)

Therefore, I will remove the value below 0 and > 600 mins.

  • DISTANCE

Negative Distances: At least one record shows –2,000 miles, clearly a data error.

Excessive Distances: Several entries fall between 8,000–10,000 miles, far beyond the maximum U.S. domestic span (~3,000 miles).

  • Dep_Delay

About 13.2% of flights (241,437 records) lie outside the whiskers, indicating extreme delays.

Some delays exceed 2,000 minutes (33+ hours), likely due to weather, mechanical issues, or cancellations.

Rather than deleting these, I’ll add a binary flag DELAY_OUTLIER so we can analyze them separately—long delays can reflect genuine operational disruptions.

  • Itin_Fare

One fare spikes around $38,400, well above typical ticket prices. Possible explanations include premium cabins, complex multi-leg itineraries, or data entry errors.

I’ll retain this record but flag any fare > $5,000 with FARE_OUTLIER and pass those to the business team for validation instead of deleting them.

  • Passengers

Passengers cannot exceed 200 per flight, as the maximum capacity is 200 people, so records with more than 200 passengers should be filtered out.

Missing Value Handling Strategy:

Based on the above graphs:

  1. For Flights Dataset

    • AIR_TIME and DISTANCE: Group by ROUTE_ID and fill with group median, as these are continuous numerical values and median is not affected by extreme values. If all of the number within the same route_id are null, then we deleted the missing value. If not, we fill it with group median.

    • DEP_DELAY and ARR_DELAY:

      • First check correlation between them. If there are correlation between them, then we could fill the missing value. Specifically, if one has value while other is missing, fill based on their relationship
      • If both of the value are missing within the same route_id are zero, then we fill it with 0 (assuming no delay)

    • FL_DATE: Delete rows, as imputation may cause bias in time series analysis

    • occupancy_rate: Delete rows, very low proportion and subsequent models are not sensitive to few missing values

  2. For Tickets Dataset

    • ITIN_FARE and PASSENGERS: Use median imputation for all the value within the same route_id, as missing proportion is low and these are numerical data.

  3. For Airports Dataset:

    • ELEVATION_FT, CONTINENT, ISO_COUNTRY, MUNICIPALITY: remain changed for those rows, as they are not important.
    • IATA_CODE: Delete rows, as missing proportion is very high, mainly for commercial airports

# Function to fill missing values by group median
def fillna_by_group_median(df, col, group_col=None):
    before_na = df[col].isna().sum()
    
    if group_col and group_col in df.columns:
        # Calculate medians for each group once
        group_medians = df.groupby(group_col)[col].median()
        # Use fillna with a mapping series for better performance
        df[col] = df[col].fillna(df[group_col].map(group_medians))
    
    after_group_na = df[col].isna().sum()
    if after_group_na > 0:
        df = df[df[col].notna()]
        print(f"{col}: After group median imputation, still {after_group_na} rows missing, all deleted")
    else:
        print(f"{col}: All missing values filled with group median (originally {before_na} rows missing)")
    
    return df

def calculate_profits(data):
    df = data.copy()
    
    # Calculate passengers and revenue
    df['passengers_per_flight'] = df['AVG_OCCUPANCY_RATE'] * MAX_CAPACITY
    df['total_passengers'] = df['passengers_per_flight'] * df['FLIGHT_COUNT']
    
    # Revenue calculation
    df['ticket_revenue'] = (df['AVG_FARE'] / 2) * df['total_passengers']
    df['baggage_revenue'] = 0.5 * BAGGAGE_FEE * df['total_passengers']
    df['total_revenue'] = df['ticket_revenue'] + df['baggage_revenue']
    
    # Cost calculation
    df['flight_costs'] = COST_PER_MILE * df['DISTANCE'] * df['FLIGHT_COUNT']
    
    # Airport fees
    origin_fee = df['ORIGIN_TYPE_flight'].map({'large_airport': LARGE_AIRPORT_FEE, 'medium_airport': MEDIUM_AIRPORT_FEE})
    dest_fee = df['DEST_TYPE_flight'].map({'large_airport': LARGE_AIRPORT_FEE, 'medium_airport': MEDIUM_AIRPORT_FEE})
    df['airport_fees'] = (origin_fee + dest_fee) * df['FLIGHT_COUNT']
    
    # Delay costs
    excess_dep = (df['AVG_DEP_DELAY'] - FREE_DELAY).clip(lower=0)
    excess_arr = (df['AVG_ARR_DELAY'] - FREE_DELAY).clip(lower=0)
    df['delay_costs'] = (excess_dep + excess_arr) * DELAY_COST * df['FLIGHT_COUNT']
    
    # Final profit calculation
    df['total_costs'] = df['flight_costs'] + df['airport_fees'] + df['delay_costs']
    df['profit'] = df['total_revenue'] - df['total_costs']
    df['profit_margin'] = (df['profit'] / df['total_revenue'] * 100).round(1)

    # Round-trip metrics
    df['roundtrips'] = df['FLIGHT_COUNT'] / 2
    df['profit_per_roundtrip']   = df['profit'] / df['roundtrips']

    return df

Approach 2: Use the Average profit per aircraft on a specific route within a quarter¶

  • My chosen metric:

The average profit earned by a single aircraft on a specific route within one quarter

(This accounts for flight frequency × profit per flight.)

  • Reasons: There could be multiple flights for an aircraft within a certain period. Even if a plane makes good profit on a single flight, that doesn’t account for the frequency of flights within the period

  • Three layers of thinking for this question:

    1. Average profit per flight for a single aircraft:
    • This doesn’t account for how often the plane flies.
    • A plane might earn a lot on one flight, but if the turnaround time is long, its total profit over a period could still be low. So this view doesn’t reflect operational efficiency or true profitability.

    2. Average profit per aircraft on a route within a quarter

    • This layer is more balanced. It combines how much profit each flight generates and how often the plane flies.
    • For example:
      • Plane A earns $10,000 per flight and flies 20 times → $200,000 total
      • Plane B earns $20,000 per flight but only flies 5 times → $100,000 total
    • By taking both of these two factors into consideration, we could clearly see what's the goal to reach for investment purpose.

    3. Total route profit:

    • This includes the sum of all profits from flights along the route during the period.
    • A route may have many flights, and the total route profit is high, but if individual flight profits are low, the average per aircraft may still be unimpressive.
    • From a strategic perspective, we could use this metrics in the long run to help the company decide whether to increase investement in certain routes. However, it may mask individual aircraft efficiency.

def calculate_investment_score(routes_data):
    df = routes_data.copy()
    
    # Normalize to 0-100 scale 
    for col in ['profit', 'profit_margin', 'FLIGHT_COUNT', 'AVG_OCCUPANCY_RATE']:
        df[f'{col}_score'] = (df[col] - df[col].min()) / (df[col].max() - df[col].min()) * 100
    
    # For delay metrics - invert the scale (lower delay = higher score)
    for col in ['AVG_DEP_DELAY', 'AVG_ARR_DELAY']:
        df[f'{col}_score'] = 100 - ((df[col] - df[col].min()) / (df[col].max() - df[col].min()) * 100)
    
    # Investment score calculation
    df['investment_score'] = (
        (df['profit_score'] * 0.3 + df['profit_margin_score'] * 0.2) +
        (df['FLIGHT_COUNT_score'] * 0.15 + df['AVG_OCCUPANCY_RATE_score'] * 0.15) +
        (df['AVG_DEP_DELAY_score'] * 0.1 + df['AVG_ARR_DELAY_score'] * 0.1)
    )
    return df
